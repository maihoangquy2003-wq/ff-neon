async function checkUID() { const inputID = document.getElementById("uid_input").value; const downloadBtn = document.getElementById("download_btn"); const infoDisplay = document.getElementById("info_display"); // Nơi hiện thời gian try { const response = await fetch('https://raw.githubusercontent.com/maihoangquy2003-wq/ff-neon/main/member.json'); const data = await response.json(); const userData = data.users[inputID]; if (!userData) { alert("ID không tồn tại trên hệ thống!"); return; } // 1. Kiểm tra xem có bị Band không if (userData.status === "banned") { alert("ID của bạn đã bị khóa khỏi hệ thống!"); downloadBtn.style.display = "none"; return; } // 2. Kiểm tra ngày hết hạn const now = new Date(); const expiryDate = new Date(userData.expiry); if (now > expiryDate) { alert("ID của bạn đã hết hạn sử dụng!"); downloadBtn.style.display = "none"; } else { // 3. Nếu hợp lệ, hiện thời gian và nút tải alert("Xác nhận thành công!"); infoDisplay.innerHTML = "Hạn dùng đến: " + userData.expiry; downloadBtn.style.display = "block"; } } catch (error) { console.log(error); alert("Lỗi kết nối máy chủ!"); } }
